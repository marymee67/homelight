<script>
    const allProducts = {
        1: products,
        2: productsPage2
    };

    let currentPage = 1;
    let currentSearchTerm = "";

    function renderCatalog() {
        let productsList = [...allProducts[currentPage]];
        
        if (currentSearchTerm.trim() !== "") {
            const searchLower = currentSearchTerm.trim().toLowerCase();
            productsList = productsList.filter(product => 
                product.name.toLowerCase().includes(searchLower) || 
                product.category.toLowerCase().includes(searchLower)
            );
        }

        const grid = document.getElementById("catalogGrid");
        
        if (productsList.length === 0) {
            grid.innerHTML = `<div class="no-results">😕 Ничего не найдено. Попробуйте другой запрос.</div>`;
            return;
        }

        grid.innerHTML = productsList.map(product => `
            <div class="card" onclick="window.location.href='card.html?id=${product.id}'">
                <div class="card-img">
                    <img src="${product.img}" alt="${product.name}">
                </div>
                <p>${product.name}</p>
                <button onclick="event.stopPropagation(); window.location.href='card.html?id=${product.id}'">перейти</button>
            </div>
        `).join("");
    }

    function updateActivePage(page) {
        const paginationSpans = document.querySelectorAll(".pagination span");
        paginationSpans.forEach(span => {
            const spanPage = parseInt(span.getAttribute("data-page"));
            if (spanPage === page) {
                span.classList.add("active");
            } else {
                span.classList.remove("active");
            }
        });
        
        const url = new URL(window.location.href);
        url.searchParams.set("page", page);
        window.history.pushState({}, "", url);
    }

    function changePage(page) {
        if (page === currentPage) return;
        currentPage = page;
        updateActivePage(currentPage);
        renderCatalog();
    }

    document.querySelectorAll(".pagination span").forEach(span => {
        span.addEventListener("click", (e) => {
            const page = parseInt(span.getAttribute("data-page"));
            changePage(page);
        });
    });

    const searchInput = document.getElementById("searchInput");
    let searchTimeout;
    
    searchInput.addEventListener("input", (e) => {
        clearTimeout(searchTimeout);
        searchTimeout = setTimeout(() => {
            currentSearchTerm = e.target.value;
            renderCatalog();
        }, 300);
    });
    
    searchInput.addEventListener("keypress", function(e) {
        if (e.key === 'Enter') {
            const query = this.value.trim().toLowerCase();
            if (query === 'доставка') window.location.href = 'delivery.html';
            else if (query === 'о нас') window.location.href = 'about.html';
            else if (query === 'главная') window.location.href = 'index.html';
            else if (query === 'корзина') window.location.href = 'trashbin.html';
            else if (query === 'профиль') window.location.href = 'profile.html';
        }
    });

    const urlParams = new URLSearchParams(window.location.search);
    const pageParam = parseInt(urlParams.get("page"));
    if (pageParam === 2) {
        currentPage = 2;
        updateActivePage(2);
    }
    
    renderCatalog();
</script>
