<script>
    function renderFavorites() {
        const items = favorites.getItems();
        const grid = document.getElementById('favoritesGrid');
        const emptyDiv = document.getElementById('emptyFavorites');
        
        if (items.length === 0) {
            grid.innerHTML = '';
            emptyDiv.style.display = 'block';
            return;
        }
        
        emptyDiv.style.display = 'none';
        
        grid.innerHTML = items.map(item => `
            <div class="favorite-card" onclick="window.location.href='card.html?id=${item.id}'">
                <button class="remove-fav-btn" onclick="event.stopPropagation(); removeFromFavorites(${item.id})">
                    <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" width="16" height="16">
                        <line x1="18" y1="6" x2="6" y2="18"/>
                        <line x1="6" y1="6" x2="18" y2="18"/>
                    </svg>
                </button>
                <div class="favorite-card-img">
                    <img src="${item.img}" alt="${item.name}">
                </div>
                <p>${item.name}</p>
                <div class="favorite-card-price">${item.price.toLocaleString('ru-RU')} ₽</div>
                <button onclick="event.stopPropagation(); addToCart(${item.id})">В КОРЗИНУ</button>
            </div>
        `).join('');
    }
    
    function removeFromFavorites(id) {
        favorites.removeFromFavorites(id);
        renderFavorites();
        showNotification('Удалено из избранного');
    }
    
    function addToCart(id) {
        const allProducts = [...products, ...productsPage2];
        const product = allProducts.find(p => p.id === id);
        if (product) {
            cart.addItem({
                id: product.id,
                name: product.name,
                price: product.price,
                img: product.img,
                category: product.category
            });
            showNotification('Товар добавлен в корзину!');
        }
    }
    
    function showNotification(message) {
        const notification = document.createElement('div');
        notification.className = 'cart-notification';
        notification.textContent = message;
        document.body.appendChild(notification);
        
        setTimeout(() => {
            notification.classList.add('show');
        }, 10);
        
        setTimeout(() => {
            notification.classList.remove('show');
            setTimeout(() => {
                notification.remove();
            }, 300);
        }, 2000);
    }
    
    const searchInput = document.getElementById('searchInput');
    if (searchInput) {
        searchInput.addEventListener('keypress', function(e) {
            if (e.key === 'Enter') {
                const query = this.value.trim().toLowerCase();
                if (query === 'каталог') window.location.href = 'catalog.html';
                else if (query === 'о нас') window.location.href = 'about.html';
                else if (query === 'главная') window.location.href = 'index.html';
                else if (query === 'корзина') window.location.href = 'trashbin.html';
                else if (query === 'избранное') window.location.href = 'favorites.html';
                else alert('По запросу "' + query + '" ничего не найдено');
            }
        });
    }
    
    renderFavorites();
</script>
