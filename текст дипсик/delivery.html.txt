<script>
    // Поиск по страницам
    const searchInput = document.getElementById('searchInput');
    if (searchInput) {
        searchInput.addEventListener('keypress', function(e) {
            if (e.key === 'Enter') {
                const query = this.value.trim().toLowerCase();
                if (query === 'каталог') window.location.href = 'catalog.html';
                else if (query === 'доставка') window.location.href = 'delivery.html';
                else if (query === 'о нас') window.location.href = 'about.html';
                else if (query === 'главная') window.location.href = 'index.html';
                else if (query === 'корзина') window.location.href = 'trashbin.html';
                else alert('По запросу "' + query + '" ничего не найдено');
            }
        });
    }

    // Эффект для сердечка
    document.querySelectorAll('.icon-favorite').forEach(icon => {
        const svg = icon.querySelector('.heart-icon');
        if (svg) {
            icon.addEventListener('mouseenter', function() {
                svg.style.stroke = '#FF6161';
                svg.style.fill = '#FF6161';
            });
            icon.addEventListener('mouseleave', function() {
                svg.style.stroke = 'currentColor';
                svg.style.fill = 'none';
            });
        }
    });

    // FAQ toggle function
    window.toggleFaq = function(element) {
        const item = element.parentElement;
        const isActive = item.classList.contains('active');
        
        document.querySelectorAll('.faq-item').forEach(faqItem => {
            faqItem.classList.remove('active');
        });
        
        if (!isActive) {
            item.classList.add('active');
        }
    };
</script>
